5 //! Simulates a system with telemerty.
6 //! Telemetry is raw data that we want to visualize to better understand its
7 //! meaning. Samples are taken at a periodic rate to animate and thus observe
8 //! changes.
9 //! The "simSystemWithTelemetry" is an imaginary system with telemetry data.
10 //! There are 8 threads with 8 entryies per thread, thus 64 32-bit
11 //! entries per sample(frame).
12 //! There are a total of 64 samples(frames) taken.
13 //! To generate a color we count the #'s of 1's(bits) in a 32-bit entry,
14 //! thus color will increase over time as the 1's are increased.

38 //! The thread function to do some simulated work.
39 //! \param id - Unique for each thread to index shared memory.
40 void tWork( uint32_t id )
41 {
42  uint32_t value = (0x01 << id);
43 
44  uint32_t *startPtr = w.getMapPtr()
45  + (id * telemetry::kNumEntriesPerThread);
46  uint32_t *ptr = startPtr;//NULL;
47  uint32_t frameCtr = 0;
48  uint32_t entryCtr = 0;
49 
50  while ( !ready._M_i ) {
51  std::this_thread::yield();
52  }
53  while ( frameCtr < telemetry::kMaxFrames ) {
54  *ptr++ = value;
55  if ((0x10000000 & value) == 0x10000000) {
56  value = (0x01 << id);
57  } else {
58  value = (value << 1) | 0x01;
59  }
60  if (++entryCtr == telemetry::kNumEntriesPerThread) {
61  startPtr += (telemetry::kNumEntriesPerFrame * telemetry::kNumEntriesPerThread);
62  ptr = startPtr;
63  } else {
64  ptr += telemetry::kNumEntriesPerFrame;
65  }
66 
67  sleepms( kThreadDelay );
68  ++frameCtr;
69  }
70 }

72 //! Simulate a system by spawning threads and setting bits in shared memory.
73 void simSystem( void )

75  cout << "Hello World!" << endl;

77  list<std::thread> tList {};
78 
79  ready.clear();

82  tList.clear();
83  for (tIdx = 0; tIdx < telemetry::kNumThreads; ++tIdx) {
84  tList.emplace_back( tWork, tIdx );
85  }
86  sleepms( 100 ); // Give the threads time to spawn.
87  cout << "Go!\n";
88  ready.test_and_set();
89  for (uint32_t i=0; i < telemetry::kMaxFrames; ++i)
90  {
91  // Allow for threads to run and fill a frame.
92  sleepms( kThreadDelay );
93  }
94  complete.test_and_set();
95  for (auto& t: tList)
96  {
97  t.join();
98  }
99 }
